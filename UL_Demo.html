  <script>

    // ══════════════════════════════════════════════════

    //  YOUR CLIENT ID

    // ══════════════════════════════════════════════════

    var CLIENT_ID = "809b4413ccd011b2a4c1b43e065bd77b";

    var EXPANDED_WIDTH = 400;

    var EXPANDED_HEIGHT = 520;

    var GR_URL_PATTERN = /\/inventory\/(\d+).*from_page=receipts/;


    var urlParams = new URLSearchParams(window.location.search);

    var iframeId = urlParams.get("floating_iframe_id") || "1";


    var currentGrNumber = null;

    var currentPoNumber = null;


    // ══════════════════════════════════════════════════

    //  IN-WIDGET DEBUG LOGGER

    //  (so you can see logs WITHOUT opening F12)

    // ══════════════════════════════════════════════════

    function debugLog(emoji, msg, data) {

      var text = emoji + " " + msg;

      if (data !== undefined) {

        try { text += " " + JSON.stringify(data); } catch(e) { text += " [object]"; }

      }

      console.log(text);


      var panel = document.getElementById("debug-panel");

      if (panel) {

        var line = document.createElement("div");

        line.className = "log-line";

        line.textContent = new Date().toLocaleTimeString() + " " + text;

        panel.appendChild(line);

        panel.scrollTop = panel.scrollHeight;

      }

    }


    function toggleDebug() {

      var panel = document.getElementById("debug-panel");

      panel.style.display = panel.style.display === "none" ? "block" : "none";

    }


    // ══════════════════════════════════════════════════

    //  IFRAME COMMUNICATION

    // ══════════════════════════════════════════════════

    debugLog("🟢", "Script loaded");

    debugLog("🟢", "Client ID:", CLIENT_ID);

    debugLog("🟢", "iFrame ID:", iframeId);

    debugLog("🟢", "URL params:", window.location.search);


    function postToParent(action, data) {

      debugLog("📤", "Sending: " + action, data);

      window.parent.postMessage(

        { action: action, client_id: CLIENT_ID, iframe_id: iframeId, data: data || {} },

        "*"

      );

    }


    function showWidget() {

      debugLog("👁️", "SHOWING widget");

      document.getElementById("app-content").style.display = "block";

      postToParent("resize", { width: EXPANDED_WIDTH, height: EXPANDED_HEIGHT, initial: false });

    }


    function hideWidget() {

      debugLog("🙈", "HIDING widget");

      postToParent("resize", { width: 0, height: 0, initial: false });

    }


    function handlePageContext(url) {

      debugLog("🔍", "Checking URL:", url);


      var match = url.match(GR_URL_PATTERN);

      debugLog("🔍", "Regex match:", match ? "YES" : "NO");


      if (match) {

        currentGrNumber = match[1];

        currentPoNumber = "PO-" + currentGrNumber.slice(0, 4);

        debugLog("✅", "GR# found:", currentGrNumber);

        updateDisplay();

        showWidget();

      } else {

        debugLog("❌", "No match — hiding widget");

        hideWidget();

      }

    }


    // ══════════════════════════════════════════════════

    //  LISTEN TO ALL MESSAGES

    // ══════════════════════════════════════════════════

    window.addEventListener("message", function (event) {

      var msg = event.data;


      // Skip non-object messages (browser noise)

      if (!msg || typeof msg !== "object") return;


      debugLog("📩", "Message from " + event.origin + ":", msg);


      if (!msg.action) {

        debugLog("⚪", "No action field — skipping");

        return;

      }


      switch (msg.action) {

        case "initIframeResponse":

        case "pageContextResponse":

          debugLog("🟡", "Got " + msg.action);


          // Try every possible URL field name

          var url = null;

          if (msg.data) {

            url = msg.data.current_url

              || msg.data.currentUrl

              || msg.data.url

              || msg.data.page_url

              || msg.data.location

              || msg.data.href

              || msg.data.pageUrl;

          }


          if (url) {

            handlePageContext(url);

          } else {

            debugLog("⚠️", "No URL found in response. Full data:", msg.data);

            // Last resort: try to get URL from the parent

            debugLog("🔄", "Trying to read parent location...");

          }

          break;


        case "locationGone":

          debugLog("🟡", "Location gone — hiding");

          hideWidget();

          break;


        default:

          debugLog("🔵", "Other action: " + msg.action);

          break;

      }

    });


    // ══════════════════════════════════════════════════

    //  INITIALIZATION

    // ══════════════════════════════════════════════════

    debugLog("📤", "Requesting initial page context...");

    postToParent("getPageContext");


    // Also try initIframe — some Coupa versions use this

    postToParent("initIframe");


    // Poll for SPA navigation

    setInterval(function () {

      postToParent("getPageContext");

    }, 3000);


    // ══════════════════════════════════════════════════

    //  DEMO FALLBACK

    //  If no Coupa response after 4 seconds,

    //  auto-activate with demo data so you can showcase

    // ══════════════════════════════════════════════════

    setTimeout(function () {

      if (!currentGrNumber) {

        debugLog("🟠", "No Coupa response after 4s — activating DEMO MODE");

        currentGrNumber = "128560";

        currentPoNumber = "PO-4500032847";

        updateDisplay();

      }

    }, 4000);


    // ══════════════════════════════════════════════════

    //  UI LOGIC

    function updateDisplay() {

      document.getElementById("gr-number").textContent = currentGrNumber || "—";

      document.getElementById("po-number").textContent = currentPoNumber || "—";

      var refs = document.querySelectorAll(".gr-ref");

      for (var i = 0; i < refs.length; i++) {

        refs[i].textContent = currentGrNumber || "—";

      }

    }


    function startCheck() {

      var btn = document.getElementById("check-btn");

      var progress = document.getElementById("progress");

      var bar = document.getElementById("progress-bar");


      btn.disabled = true;

      btn.innerHTML = '<div class="spinner"></div> Checking...';

      progress.style.display = "block";

      document.getElementById("result-success").style.display = "none";

      document.getElementById("result-error").style.display = "none";

      document.getElementById("reset-link").style.display = "none";


      for (var i = 1; i <= 4; i++) {

        document.getElementById("step-" + i).className = "progress-step";

        document.getElementById("icon-" + i).className = "step-icon pending";

        document.getElementById("icon-" + i).innerHTML = "";

      }


      var steps = [

        { delay: 600,  step: 1, progress: 25  },

        { delay: 1800, step: 2, progress: 50  },

        { delay: 3200, step: 3, progress: 75  },

        { delay: 4500, step: 4, progress: 100 }

      ];


      steps.forEach(function (s) {

        setTimeout(function () {

          document.getElementById("step-" + s.step).className = "progress-step active";

          document.getElementById("icon-" + s.step).className = "step-icon active";

          document.getElementById("icon-" + s.step).innerHTML = '<div class="spinner" style="width:12px;height:12px;border-width:2px;"></div>';

          bar.style.width = (s.progress - 15) + "%";

        }, s.delay);


        setTimeout(function () {

          document.getElementById("step-" + s.step).className = "progress-step done";

          document.getElementById("icon-" + s.step).className = "step-icon done";

          document.getElementById("icon-" + s.step).innerHTML = "✓";

          bar.style.width = s.progress + "%";

        }, s.delay + 800);

      });


      setTimeout(function () {

        var simulateFail = document.getElementById("demo-fail-toggle").checked;


        if (simulateFail) {

          document.getElementById("result-error").style.display = "block";

        } else {

          document.getElementById("sap-invoice-success").textContent =

            "51" + String(Math.floor(Math.random() * 9000000 + 1000000));

          document.getElementById("invoice-amount").textContent =

            "€ " + (Math.floor(Math.random() * 50000) + 5000).toLocaleString("de-DE") + ",00";

          document.getElementById("posting-date").textContent =

            new Date(Date.now() - Math.random() * 7 * 86400000)

              .toLocaleDateString("de-DE");

          document.getElementById("result-success").style.display = "block";

        }


        document.getElementById("reset-link").style.display = "block";

        btn.disabled = false;

        btn.innerHTML = "⚡ Check Invoice Status in ERP";

      }, 5800);

    }

      }

    }
